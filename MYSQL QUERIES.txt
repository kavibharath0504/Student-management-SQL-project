SELECT * FROM ENROLLMENTS;

SELECT* FROM STUDENTS ORDER BY NAME ;
SELECT COUNT(*) AS TOTAL_ENROLLMENTS FROM ENROLLMENTS;
SELECT COUNT(*) AS TOTAL_STUDENTS FROM STUDENTS;
SELECT NAME,PHONE FROM STUDENTS WHERE NAME LIKE'KAVI';

SELECT S.NAME,C.COURSE_NAME,E.ENROLL_ID FROM STUDENTS S
JOIN ENROLLMENTS E ON S.STUDENT_ID=E.STUDENT_ID 
JOIN COURSES C ON E.COURSE_ID= C.COURSE_ID;

SELECT S.NAME,C.COURSE_NAME,M.SCORE FROM MARKS M
JOIN STUDENTS S ON M.STUDENT_ID=S.STUDENT_ID 
JOIN COURSES C ON M.COURSE_ID= C.COURSE_ID;

SELECT C.COURSE_NAME, MAX(M.SCORE) AS HIGHEST_SCORE
FROM MARKS M JOIN COURSES C ON M.COURSE_ID =C.COURSE_ID
GROUP BY C.COURSE_ID;

SELECT S.NAME,AVG (M.SCORE) AS AVG_SCORE
FROM MARKS M JOIN STUDENTS S ON M.STUDENT_ID=S.STUDENT_ID
GROUP BY S.NAME;

SELECT S.NAME,M.SCORE FROM MARKS M 
JOIN STUDENTS S ON S.STUDENT_ID=M.STUDENT_ID
ORDER BY M.SCORE LIMIT 2;

SELECT* FROM MARKS WHERE SCORE=(SELECT MAX(SCORE)FROM MARKS);
